### Para filtrar se usa FILTER

```js
function esPar(nro){
	return nro % 2 == 0;
}

console.log(numeros.filter(esPar));
```

### Para transformar se usa MAP

```js
numeros = [2,5,8,9,12,6,7]

function doble (nro){
	var elDoble = nro*2;
	return elDoble;
  }

numeros.map(doble)
```
